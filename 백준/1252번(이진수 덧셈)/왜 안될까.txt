#include <stdio.h>
#include <math.h>

int Dec(int num){
	
	int arr[80] = {0,};
	int i = 0, sum = 0;
	
	while(num != 0){
		arr[i] = num % 10;
		num /= 10;
		i++;
	}

	for(int j=0; j<i; j++){
		int temp = pow(2,j);
		if(arr[j] == 1){
			sum += temp;
		}
	}
	
	return sum;	
}
